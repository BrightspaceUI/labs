The following demos all use the same ReactiveStore definition, which is shown below: import ReactiveStore from '../../../src/utilities/reactive-store/reactive-store.js'; export default class MyStore extends ReactiveStore { static get properties() { return { foo: { type: Number }, bar: { type: Number }, }; } constructor() { super(); this.foo = 0; this.bar = 0; } }

Basic Demo

This demo defines a single component that connects to a ReactiveStore instance using the store consumer controller. See the definition for this component here: import { html, LitElement } from 'lit'; import MyStore from './my-store.js'; // Create a new instance of MyStore // Note that this would normally be defined in a separate file so that it can be imported by other components const myStore = new MyStore(); const MyStoreConsumer = myStore.createConsumer(); class BasicDemo extends LitElement { constructor() { super(); this.consumer = new MyStoreConsumer(this); } render() { return html` <div>Foo: ${this.consumer.foo}</div> <div>Bar: ${this.consumer.bar}</div> <button @click=${this._updateFoo}>Update foo</button> <button @click=${this._updateBar}>Update bar</button> `; } _updateBar() { this.consumer.bar += 1; } _updateFoo() { this.consumer.foo += 1; } } customElements.define('basic-demo', BasicDemo);

Multiple Consumers Demo

This demo defines two components (one parent and one child) that both connect to the same ReactiveStore instance using the store consumer controller. See the definition for both of these components here: import { html, LitElement } from 'lit'; import MyStore from './my-store.js'; // Create a new instance of MyStore // Note that this would normally be defined in a separate file so that it can be imported by other components const myStore = new MyStore(); const MyStoreConsumer = myStore.createConsumer(); class MultipleConsumersDemo extends LitElement { constructor() { super(); this.consumer = new MyStoreConsumer(this); } render() { return html` <h2>Parent</h2> <div>Foo: ${this.consumer.foo}</div> <div>Bar: ${this.consumer.bar}</div> <button @click=${this._click}>Update foo</button> <multiple-consumers-demo-child></multiple-consumers-demo-child> `; } _click() { this.consumer.foo += 1; } } class MultipleConsumersDemoChild extends LitElement { constructor() { super(); this.consumer = new MyStoreConsumer(this); } render() { return html` <h2>Child</h2> <div>Foo: ${this.consumer.foo}</div> <div>Bar: ${this.consumer.bar}</div> <button @click=${this._click}>Update bar</button> `; } _click() { this.consumer.bar += 1; } } customElements.define('multiple-consumers-demo-child', MultipleConsumersDemoChild); customElements.define('multiple-consumers-demo', MultipleConsumersDemo);

Context Demo

This demo defines two components: a parent and a child. The parent instantiates and provides an instance of MyStore by using the store context Provider controller. The child then consumes the store using the store context Consumer controller. See the definition for both of these components here: import { html, LitElement } from 'lit'; import MyStore from './my-store.js'; // Define the context controllers // Note that this would normally be defined in a separate file so that it can be imported by other components const { Provider: MyStoreContextProvider, Consumer: MyStoreContextConsumer } = MyStore.createContextControllers(); class ContextDemo extends LitElement { constructor() { super(); this.store = new MyStoreContextProvider(this); } render() { return html` <h2>Parent</h2> <div>Foo: ${this.store.foo}</div> <div>Bar: ${this.store.bar}</div> <button @click=${this._click}>Update foo</button> <context-demo-child></context-demo-child> `; } _click() { this.store.foo += 1; } } class ContextDemoChild extends LitElement { constructor() { super(); this.consumer = new MyStoreContextConsumer(this); } render() { return html` <h2>Child</h2> <div>Foo: ${this.consumer.foo}</div> <div>Bar: ${this.consumer.bar}</div> <button @click=${this._click}>Update bar</button> `; } _click() { this.consumer.bar += 1; } } customElements.define('context-demo-child', ContextDemoChild); customElements.define('context-demo', ContextDemo);
