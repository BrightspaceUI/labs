The following demos all use one of the ReactiveStore definitions shown below: import ReactiveStore from '../../../src/utilities/reactive-store/reactive-store.js'; export default class MyStore extends ReactiveStore { static get properties() { return { count: { type: Number }, }; } constructor() { super(); this.count = 0; } decrement() { this.count--; } increment() { this.count++; } } import ReactiveStore from '../../../src/utilities/reactive-store/reactive-store.js'; export default class MultiPropStore extends ReactiveStore { static get properties() { return { foo: { type: Number }, bar: { type: Number }, baz: { type: Number }, }; } constructor() { super(); this.foo = 0; this.bar = 0; this.baz = 0; } }

Basic Demo

This demo defines a single component that connects to a ReactiveStore instance using the store consumer controller. See the definition for this component here: import { html, LitElement } from 'lit'; import MyStore from './my-store.js'; // Create a new instance of MyStore // Note that this would normally be defined in a separate file so that it can be imported by other components const myStore = new MyStore(); const MyStoreConsumer = myStore.createConsumer(); class BasicDemo extends LitElement { constructor() { super(); this.consumer = new MyStoreConsumer(this); } render() { return html` <div>Count: ${this.consumer.count}</div> <button @click=${this._increment}>Increment</button> <button @click=${this._decrement}>Decrement</button> <button @click=${this._reset}>Reset</button> `; } _decrement() { this.consumer.decrement(); } _increment() { this.consumer.increment(); } _reset() { this.consumer.count = 0; } } customElements.define('basic-demo', BasicDemo);

Context Demo

This demo defines two components: a parent and a child. The parent instantiates and provides an instance of MyStore by using the store context Provider controller. The child then consumes the store using the store context Consumer controller. See the definition for both of these components here: import { html, LitElement } from 'lit'; import MyStore from './my-store.js'; // Define the context controllers // Note that this would normally be defined in a separate file so that it can be imported by other components const { Provider: MyStoreContextProvider, Consumer: MyStoreContextConsumer } = MyStore.createContextControllers(); class ContextDemo extends LitElement { constructor() { super(); this.store = new MyStoreContextProvider(this); } render() { return html` <h2>Parent</h2> <div>Count: ${this.store.count}</div> <button @click=${this._increment}>Increment</button> <button @click=${this._decrement}>Decrement</button> <button @click=${this._reset}>Reset</button> <context-demo-child></context-demo-child> `; } _decrement() { this.store.decrement(); } _increment() { this.store.increment(); } _reset() { this.store.count = 0; } } class ContextDemoChild extends LitElement { constructor() { super(); this.consumer = new MyStoreContextConsumer(this); } render() { return html` <h2>Child</h2> <div>Count: ${this.consumer.count}</div> <button @click=${this._increment}>Increment</button> <button @click=${this._decrement}>Decrement</button> <button @click=${this._reset}>Reset</button> `; } _decrement() { this.consumer.decrement(); } _increment() { this.consumer.increment(); } _reset() { this.consumer.count = 0; } } customElements.define('context-demo-child', ContextDemoChild); customElements.define('context-demo', ContextDemo);

Individual Property Reactivity Demo

This demo uses a store with 3 properties (foo, bar and baz) and defines 2 components:
1. A simple component with buttons to update the store properties.
2. A component that consumes the store, displays the foo and bar properties (but not baz), and is instantiated 3 separate times with different properties.

The consuming component also has the option to pass the dependentProperties and detectDependentProperties options to the store consumer controller. Additionally, it displays how many times its render cycle has been executed.

See the definition for both of these components here:

import { html, LitElement } from 'lit'; import MultiPropStore from './multi-prop-store.js'; // Create a new instance of MultiPropStore // Note that this would normally be defined in a separate file so that it can be imported by other components const multiPropStore = new MultiPropStore(); const MultiPropStoreConsumer = multiPropStore.createConsumer(); // This component is only used to show the store state and update its properties class IndividualReactivityStoreControls extends LitElement { constructor() { super(); this.consumer = new MultiPropStoreConsumer(this); } render() { return html` <div>Foo: ${this.consumer.foo}</div> <button @click=${this._incrementFoo}>Increment Foo</button> <div>Bar: ${this.consumer.bar}</div> <button @click=${this._incrementBar}>Increment Bar</button> <div>Baz: ${this.consumer.baz}</div> <button @click=${this._incrementBaz}>Increment Baz</button> `; } _incrementBar() { this.consumer.bar++; } _incrementBaz() { this.consumer.baz++; } _incrementFoo() { this.consumer.foo++; } } customElements.define('individual-reactivity-store-controls', IndividualReactivityStoreControls); // This component is used to show the individual reactivity features of the store consumer class IndividualReactivityConsumingComponent extends LitElement { static properties = { // This property is directly passed to the store consumer's dependentProperties option dependentProperties: { type: Array, attribute: 'dependent-properties', converter: (value) => (value ? value.split(',') : undefined), }, // This property is directly passed to the store consumer's detectDependentProperties option detectDependentProperties: { type: Boolean, attribute: 'detect-dependent-properties' }, }; constructor() { super(); this.renderCount = 0; } connectedCallback() { super.connectedCallback(); this.consumer = new MultiPropStoreConsumer(this, { // The store consumer options are passed in from the component's own properties for the // sake of the demo, but in a real use case they would likely be hardcoded values. dependentProperties: this.dependentProperties, detectDependentProperties: this.detectDependentProperties, }); } render() { this.renderCount++; // From the store, we only render the foo and bar properties. The baz property is not used // by this component. // A render count is also included to show when the component re-renders. return html` <div>Foo: ${this.consumer.foo}</div> <div>Bar: ${this.consumer.bar}</div> <div>Render Count: ${this.renderCount}</div> `; } } customElements.define('individual-reactivity-consuming-component', IndividualReactivityConsumingComponent);

Since the consuming components only use the foo and bar properties from the store, they should theoretically only need to re-render when those properties change and not when baz changes. However, by default, the consuming components are dependent on all properties from the store, so they will re-render when baz changes even though they don't use it.

In order to avoid unnecessary re-renders, we have two options:
1. Explicitly specify dependent properties by passing the dependentProperties option to the store consumer controller. This is shown in the second consuming component in the demo, which is passing dependant properties to the consumer via dependent-properties="foo,bar".
2. Enable automatic detection of dependent properties by passing the detectDependentProperties option to the store consumer controller. This is shown in the third consuming component in the demo, which is enabling dependency detection via the detect-dependent-properties attribute.

With these options, we can ensure that the consuming components only re-render when the properties they actually depend on change.
